91 int EGS_cSpheres::inside(const EGS_Vector &x) {
92 
93  EGS_Vector tmp(x-xo);
94  EGS_Float r_sq=tmp.length2();
95 
96  // are we outside off all spheres? If so return that region number
97  if (r_sq>R2[nreg-1]) {
98  return -1;
99  }
100 
101  /*
102  * algorithm below fails for particle in central sphere ...ugh!
103  */if (r_sq<R2[0]) {
104  return 0;
105  }
106 
107  // search for region containing particle
108  int is=0,os=nreg,ms;
109  while (os-is>1) {
110  ms=(is+os)/2;
111  if (r_sq<=R2[ms]) {
112  os=ms;
113  }
114  else {
115  is=ms;
116  }
117  }
118  return os;

133 EGS_Float EGS_cSpheres::howfarToOutside(int ireg, const EGS_Vector &x,
134  const EGS_Vector &u) {
135  if (ireg < 0) {
136  return 0;
137  }
138  EGS_Vector xp(x - xo);
139  EGS_Float aa = xp*u, aa2 = aa*aa;
140  EGS_Float bb2 = xp.length2();
141  EGS_Float R2b2 = R2[nreg-1] - bb2;
142  if (R2b2 <= 0) {
143  return 0; // outside within precision
144  }
145  EGS_Float tmp = sqrt(aa2 + R2b2);
146  return aa > 0 ? R2b2/(tmp + aa) : tmp - aa;
147 }

149 int EGS_cSpheres::howfar(int ireg, const EGS_Vector &x,
150  const EGS_Vector &u, EGS_Float &t, int *newmed, EGS_Vector *normal) {
151  int direction_flag=-1; /* keep track of direction entering or exiting a
152  sphere boundary */
153  double d=veryFar*1e5; // set a maximum distance from a boundary
154 
155  EGS_Vector xp(x - xo);
156  double aa = xp*u, aa2 = aa*aa;
157  double bb2 = xp.length2();
158 
159  double rad=0, R2b2, tmp;
160 
161  // check if we are inside of any regions at all? ...
162  if (ireg>=0) {
163 
164  /* check if particle is moving towards or away from the sphere(s) centre.
165  * we loose here if the particle is in the centre sphere as we don't need
166  * to check this condition. see next 'if' statement
167  */
168  if (aa >= 0 || !ireg) {
169 
170  /* ie. particle moving away from center of
171  spheres, OR it is IN the innermost sphere
172  => we must check the outer sphere only
173  */
174  R2b2 = R2[ireg] - bb2;
175  if (R2b2 <= 0 && aa > 0) {
176  d = boundaryTolerance; // hopefully a truncation problem
177  }
178  else {
179  tmp = aa2 + R2b2;
180  if (tmp > 0) {
181  tmp = sqrt(tmp);
182  }
183  else {
184  if (tmp < -boundaryTolerance) {
185  egsWarning("EGS_cSpheres::howfar: something is wrong\n");
186  egsWarning(" we think we are in region %d, but R2b2=%g",
187  ireg,R2b2);
188  }
189  tmp = 0;
190  }
191  d = aa > 0 ? R2b2/(tmp + aa) : tmp - aa;
192  // the above reduces roundoff, which is significant
193  // when aa2 is large compared to R2b2 and aa>0
194  }
195  rad = -R[ireg];
196  direction_flag=ireg+1;
197  if (direction_flag >= nreg) {
198  direction_flag = -1;
199  }
200  }
201  else {
202 
203  /* so now we know the particle is moving towards the centre of the
204  * spheres. check to see if its trajectory will intersect the nested
205  * sphere - we are guaranteed there is one - we checked that already!
206  */
207  R2b2 = R2[ireg-1] - bb2;
208  tmp = aa2 + R2b2;
209  if (tmp <= 0) { // we will not intersect the nested sphere
210  R2b2 = R2[ireg] - bb2;
211  tmp = aa2 + R2b2;
212  if (tmp > 0) {
213  d = sqrt(tmp) - aa;
214  }
215  else {
216  d = -aa;
217  }
218  rad = -R[ireg];
219  direction_flag=ireg+1;
220  if (direction_flag >= nreg) {
221  direction_flag = -1;
222  }
223  }
224  else {
225  // we're hitting the inner sphere (from the outside)
226  tmp = sqrt(tmp);
227  d = -R2b2/(tmp - aa);
228  direction_flag=ireg-1;
229  rad = R[direction_flag];
230  }
231  }
232  }
233  else {
234  // we are not inside any of the spherical regions of interest
235  if (aa<0) { // we _might_ intersect the largest sphere
236  R2b2 = R2[nreg-1] - bb2;
237  tmp = aa2 + R2b2;
238  if (tmp > 0) { // we *will* intersect the largest sphere
239  d = -R2b2/(sqrt(tmp) - aa);
240  direction_flag=nreg-1;
241  rad = R[direction_flag];
242  }
243  }
244  }
245 
246 #ifdef SPHERES_DEBUG
247  if (isnan(d)) {
248  egsWarning("\nGot nan\n");
249  }
250 
251  if (d < -boundaryTolerance) {
252  egsWarning("\nNegative step?: %g\n",d);
253  egsWarning("ireg=%d inew=%d aa=%g bb2=%g\n",ireg,direction_flag,aa,bb2);
254  //exit(1);
255  }
256 
257  last_x = x;
258  last_u = u;
259  last_ireg = ireg;
260  last_d = d;
261  last_t = t;
262  last_aa = aa;
263  last_bb2 = bb2;
264  last_R2b2 = R2b2;
265  last_tmp = tmp;
266 #endif
267 
268  // check desired step size against this d
269  if (d<=t) {
270  t=d;
271  if (newmed) {
272  if (direction_flag >= 0) {
273  *newmed = medium(direction_flag);
274  }
275  else {
276  *newmed = -1;
277  }
278  }
279  if (normal) {
280  EGS_Vector n(xp + u*d);
281  *normal = n*(1/rad);
282  }
283  return direction_flag;
284  }
285  return ireg;
286 }

289 EGS_Float EGS_cSpheres::hownear(int ireg, const EGS_Vector &x) {
290  EGS_Vector xp(x-xo);
291  EGS_Float r=xp.length();
292  //EGS_Float r_sq=x.x*x.x+x.y*x.y+x.z*x.z;
293  EGS_Float d;
294 
295  if (ireg>=0) {
296  d=R[ireg]-r;
297  if (ireg) {
298  EGS_Float dd=r-R[ireg-1];
299  if (dd<d) {
300  d=dd;
301  }
302  }
303  }
304  else {
305  d=r-R[nreg-1];
306  }
307 
308  return d;
